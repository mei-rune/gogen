func InitStringSvc(mux *echo.Group, svc StringSvc) {
	mux.GET("/echo", func(ctx echo.Context) error {
		var a = ctx.QueryParam("a")

		result := svc.Echo(a)
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/echo", func(ctx echo.Context) error {

		result, err := svc.EchoBody(ctx.Request().Body)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/concat", func(ctx echo.Context) error {
		var a = ctx.QueryParam("a")
		var b = ctx.QueryParam("b")

		result, err := svc.Concat(a, b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/concat1", func(ctx echo.Context) error {
		var a *string
		if s := ctx.QueryParam("a"); s != "" {
			a = &s
		}
		var b *string
		if s := ctx.QueryParam("b"); s != "" {
			b = &s
		}

		result, err := svc.Concat1(a, b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/concat2/:a/:b", func(ctx echo.Context) error {
		var a = ctx.Param("a")
		var b = ctx.Param("b")

		result, err := svc.Concat2(a, b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/concat3/:a/:b", func(ctx echo.Context) error {
		var a = ctx.Param("a")
		var b = ctx.Param("b")

		result, err := svc.Concat3(&a, &b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/sub", func(ctx echo.Context) error {
		var a = ctx.QueryParam("a")
		var start int64
		if s := ctx.QueryParam("start"); s != "" {
			v64, err := strconv.ParseInt(s, 10, 64)
			if err != nil {
				ctx.Error(fmt.Errorf("argument %q is invalid - %q", start, s, err))
				return nil
			}
			start = v64
		}

		result, err := svc.Sub(a, start)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.POST("/save/:a", func(ctx echo.Context) error {
		var a = ctx.Param("a")
		var b string
		if err := ctx.Bind(&b); err != nil {
			ctx.Error(fmt.Errorf("argument %q is invalid - %q", b, "<no value>", err))
			return nil
		}

		result, err := svc.Save(a, b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusCreated, result)
	})
	mux.POST("/save2/:a", func(ctx echo.Context) error {
		var a = ctx.Param("a")
		var b string
		if err := ctx.Bind(&b); err != nil {
			ctx.Error(fmt.Errorf("argument %q is invalid - %q", b, "<no value>", err))
			return nil
		}

		result, err := svc.Save2(&a, &b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusCreated, result)
	})
	mux.GET("/add/:a/:b", func(ctx echo.Context) error {
		var a int
		if v64, err := strconv.ParseInt(ctx.Param("a"), 10, 64); err != nil {
			s := ctx.Param("a")
			ctx.Error(fmt.Errorf("argument %q is invalid - %q", a, s, err))
			return nil
		} else {
			a = int(v64)
		}
		var b int
		if v64, err := strconv.ParseInt(ctx.Param("b"), 10, 64); err != nil {
			s := ctx.Param("b")
			ctx.Error(fmt.Errorf("argument %q is invalid - %q", b, s, err))
			return nil
		} else {
			b = int(v64)
		}

		result, err := svc.Add(a, b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/add2/:a/:b", func(ctx echo.Context) error {
		var a *int
		if v64, err := strconv.ParseInt(ctx.Param("a"), 10, 64); err != nil {
			s := ctx.Param("a")
			ctx.Error(fmt.Errorf("argument %q is invalid - %q", a, s, err))
			return nil
		} else {
			a = new(int)
			*a = int(v64)
		}
		var b *int
		if v64, err := strconv.ParseInt(ctx.Param("b"), 10, 64); err != nil {
			s := ctx.Param("b")
			ctx.Error(fmt.Errorf("argument %q is invalid - %q", b, s, err))
			return nil
		} else {
			b = new(int)
			*b = int(v64)
		}

		result, err := svc.Add2(a, b)
		if err != nil {
			ctx.Error(err)
			return nil
		}
		return ctx.JSON(http.StatusOK, result)
	})
	mux.GET("/add3", func(ctx echo.Context) error {
		var a *int
		if s := ctx.QueryParam("a"); s != "" {
			v64, err := strconv.ParseInt(s, 10, 64)
			if err != nil {
