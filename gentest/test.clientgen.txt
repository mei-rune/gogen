type TestClient struct {
	proxy *resty.Proxy
}

func (client TestClient) Echo(ctx context.Context, a string) (string, error) {
	var result string

	request := resty.NewRequest(client.proxy, "/echo").
		SetParam("a", a).
		Result(&result)

	err := request.GET(ctx)

	resty.ReleaseRequest(client.proxy, request)
	return result, err
}
