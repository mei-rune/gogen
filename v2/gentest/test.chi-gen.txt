		queryParams := r.URL.Query()
		var filenames = queryParams["filenames"]
		list, total, err := svc.GetFiles(filenames)

			render.Status(r, httpCodeWith(err))
			render.JSON(w, r, err)
			return
		}

			"list":  list,

		}
		render.JSON(w, r, result)
		return
