func InitOptionalPrefixSvc(mux chi.Router, enabledPrefix bool, svc OptionalPrefixSvc) {
	initFunc := func(mux chi.Router) {
		mux.Get("/get", func(w http.ResponseWriter, r *http.Request) {
			err := svc.Get()
			if err != nil {
				render.Status(r, httpCodeWith(err))
				render.JSON(w, r, err)
				return
			}
			render.JSON(w, r, "OK")
			return
		})
	}
	if enabledPrefix {
		mux = mux.Route("/optpre", initFunc)
	} else {
		initFunc(mux)
	}
}
